# Code Blocks

This page demonstrates the CodeBlock component with various examples.

## Fenced Code Blocks

Fenced code blocks in markdown are automatically converted to CodeBlock components:

```javascript
function greet(name) {
  return `Hello, ${name}!`;
}

console.log(greet("World"));
```

```typescript
interface User {
  id: number;
  name: string;
  email: string;
}

function createUser(name: string, email: string): User {
  return {
    id: Date.now(),
    name,
    email,
  };
}
```

```jsx
import React from 'react';

export function Button({ children, onClick }) {
  return (
    <button onClick={onClick} className="btn">
      {children}
    </button>
  );
}
```

## Explicit CodeBlock Usage

You can also use the CodeBlock component directly with more advanced features:

### Basic Code Block

<CodeBlock
  code={`function calculateTotal(items) {
  let total = 0;
  for (const item of items) {
    total += item.price * item.quantity;
  }
  return total;
}`}
  language="javascript"
  filename="utils.js"
/>

### With Highlighted Lines

<CodeBlock
  code={`function processData(data) {
  // Validate input
  if (!data || !Array.isArray(data)) {
    throw new Error('Invalid data');
  }
  
  // Process each item
  return data.map(item => {
    return transform(item);
  });
}`}
  language="javascript"
  filename="processor.js"
  highlightLines={[2, 3, 4, 7, 8, 9]}
/>

### Multiple Tabs (Different Files)

<CodeBlock
  snippets={[
    {
      code: `export const Button = ({ children, onClick }) => {
  return (
    <button onClick={onClick} className="btn">
      {children}
    </button>
  );
};`,
      language: 'jsx',
      filename: 'Button.jsx',
      tabTitle: 'Button.jsx',
    },
    {
      code: `.btn {
  padding: 0.5rem 1rem;
  background-color: #3b82f6;
  color: white;
  border-radius: 0.375rem;
  transition: background-color 0.2s;
}

.btn:hover {
  background-color: #2563eb;
}`,
      language: 'css',
      filename: 'Button.css',
      tabTitle: 'Button.css',
    },
    {
      code: `import { Button } from './Button';

export default {
  title: 'Components/Button',
  component: Button,
};

export const Primary = {
  args: {
    children: 'Click me',
  },
};`,
      language: 'typescript',
      filename: 'Button.stories.ts',
      tabTitle: 'Button.stories.ts',
    },
  ]}
/>

### Multiple Languages

<CodeBlock
  snippets={[
    {
      code: `function greet(name: string): string {
  return \`Hello, \${name}!\`;
}`,
      language: 'typescript',
      filename: 'greet.ts',
    },
    {
      code: `function greet(name) {
  return \`Hello, \${name}!\`;
}`,
      language: 'javascript',
      filename: 'greet.js',
    },
    {
      code: `def greet(name):
    return f"Hello, {name}!"`,
      language: 'python',
      filename: 'greet.py',
    },
    {
      code: `def greet(name)
  "Hello, #{name}!"
end`,
      language: 'ruby',
      filename: 'greet.rb',
    },
  ]}
/>

## Combining with Other Components

You can combine CodeBlock with other components like Callouts:

<Callout variant="tip">
  When using CodeBlock with multiple tabs, make sure each snippet has a unique `tabTitle` or `filename` for better UX.
</Callout>

<CodeBlock
  code={`// This is a simple example
const message = "Hello, World!";
console.log(message);`}
  language="javascript"
  highlightLines={[2]}
/>
